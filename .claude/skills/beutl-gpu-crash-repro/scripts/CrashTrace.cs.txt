using System.Text;

namespace Beutl.Graphics;

internal static class CrashTrace
{
    private static readonly FileStream s_fs =
        new("/dumps/lastrestore.txt", FileMode.Create, FileAccess.Write, FileShare.ReadWrite);

    private static readonly object s_lock = new();

    public static void Mark(string tag)
    {
        lock (s_lock)
        {
            s_fs.Seek(0, SeekOrigin.Begin);
            byte[] b = Encoding.ASCII.GetBytes(tag + " tid=" + Environment.CurrentManagedThreadId + "\n");
            s_fs.Write(b, 0, b.Length);
            s_fs.SetLength(b.Length);
            // Plain Flush(), NOT Flush(true): the OS write survives process death since the kernel keeps the
            // file buffer after a segfault, and fsync-per-call is far too slow over millions of calls.
            s_fs.Flush();
        }
    }
}
